\subsection{Auditable CTGAN}

We extend the standard CTGAN with audit logging capabilities:

\begin{lstlisting}[language=Python, caption=Auditable CTGAN Structure]
class AuditableCTGAN:
    def __init__(self, epochs=100):
        self.model = CTGAN(epochs=epochs)
        self.audit_logs = []
        
    def fit(self, data, discrete_columns):
        self.input_hash = compute_hash(data)
        self.model.fit(data, discrete_columns)
        self.log_training_event()
        
    def sample(self, n_samples):
        synthetic = self.model.sample(n_samples)
        self.output_hash = compute_hash(synthetic)
        self.log_generation_event()
        return synthetic
\end{lstlisting}

\subsection{Blockchain Integration}

The blockchain client provides a unified interface for both simulation and production modes:

\begin{lstlisting}[language=Python, caption=Blockchain Client]
class BlockchainClient:
    def __init__(self, mode=BlockchainMode.SIMULATION):
        if mode == BlockchainMode.SIMULATION:
            self.backend = SimulatedBlockchain()
        else:
            self.backend = HyperledgerFabric()
            
    def log_verification(self, data_hash, results):
        transaction = {
            'data_hash': data_hash,
            'results': results,
            'timestamp': datetime.now().isoformat()
        }
        return self.backend.submit_transaction(transaction)
\end{lstlisting}

\subsection{Consensus Engine}

The consensus engine aggregates results from multiple verifiers:

\begin{lstlisting}[language=Python, caption=Consensus Engine]
class ConsensusEngine:
    def __init__(self, num_verifiers=3, threshold=70):
        self.num_verifiers = num_verifiers
        self.threshold = threshold
        
    def reach_consensus(self, verifications):
        scores = [v['overall_score'] for v in verifications]
        final_score = np.median(scores)
        status = 'approved' if final_score >= self.threshold else 'rejected'
        return ConsensusRecord(
            final_score=final_score,
            status=status,
            num_verifiers=len(verifications)
        )
\end{lstlisting}
